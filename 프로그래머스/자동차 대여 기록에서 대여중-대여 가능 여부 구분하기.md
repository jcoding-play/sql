```sql
SELECT 
    CAR_ID,
    CASE 
        WHEN MAX(AVAILABILITY) = 2 THEN "대여중"
        ELSE "대여 가능"
    END AS AVAILABILITY
FROM (
    SELECT
        CAR_ID, 
        CASE 
            WHEN '2022-10-16' BETWEEN START_DATE AND END_DATE THEN 2 # 대여중
            ELSE 1 
        END AS AVAILABILITY
    FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY
) AS T
GROUP BY CAR_ID
ORDER BY CAR_ID DESC
```
